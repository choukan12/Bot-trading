// signals.js - Module de génération de signaux simples et fiables // Calcul RSI (Relative Strength Index) - période 14 recommandée function calculateRSI(prices, period = 14) { if (prices.length < period + 1) return null; let gains = 0, losses = 0; for (let i = 1; i <= period; i++) { const change = prices[prices.length - i] - prices[prices.length - i - 1]; if (change > 0) gains += change; else losses -= change; } const avgGain = gains / period; const avgLoss = losses / period; if (avgLoss === 0) return 100; const rs = avgGain / avgLoss; return 100 - (100 / (1 + rs)); } // Calcul MACD simple (12, 26, 9) function calculateMACD(prices) { if (prices.length < 50) return { macd: null, signal: null, histogram: null }; // On utilise une version simplifiée avec EMA approximée (tu peux améliorer plus tard) const ema12 = prices.slice(-12).reduce((a, b) => a + b, 0) / 12; const ema26 = prices.slice(-26).reduce((a, b) => a + b, 0) / 26; const macd = ema12 - ema26; const signal = macd * 0.9; // approximation simple du signal line return { macd: parseFloat(macd.toFixed(4)), signal: parseFloat(signal.toFixed(4)), histogram: parseFloat((macd - signal).toFixed(4)) }; } // Génère le signal principal export function getTradingSignal(priceHistory) { if (priceHistory.length < 30) { return { signal: "WAIT", strength: 0, reason: "Pas assez de données" }; } const currentPrice = priceHistory[priceHistory.length - 1]; const rsi = calculateRSI(priceHistory); const macd = calculateMACD(priceHistory); let signal = "NEUTRAL"; let strength = 0; let reason = ""; // Logique de signal (très basique mais efficace pour commencer) if (rsi < 30 && macd.histogram > 0) { signal = "BUY"; strength = 75; reason = "RSI Oversold + MACD Bullish"; } else if (rsi > 70 && macd.histogram < 0) { signal = "SELL"; strength = 75; reason = "RSI Overbought + MACD Bearish"; } else if (rsi < 40 && macd.macd > macd.signal) { signal = "BUY"; strength = 55; reason = "RSI bas + MACD crossover haussier"; } else if (rsi > 60 && macd.macd < macd.signal) { signal = "SELL"; strength = 55; reason = "RSI haut + MACD crossover baissier"; } return { signal, strength, rsi: rsi ? parseFloat(rsi.toFixed(2)) : null, macd: macd, price: currentPrice, timestamp: new Date().toISOString(), reason }; }
